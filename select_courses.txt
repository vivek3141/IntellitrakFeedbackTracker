select c.course_name as Course, f1.fb_text Feedback1, f2.fb_text Feedback2, f3.fb_text Feedback3, f4.fb_text Feedback4, f5.fb_text Feedback5, oth_1 OtherCourses, oth_2 OtherCourses, oth_3 OtherCourses, oth_4 OtherCourses, oth_5 OtherCourses 
from courses c, feedback f1, feedback f2, feedback f3, feedback f4, feedback f5
where c.fb_id1 = f1.fb_id and c.fb_id2 =f2.fb_id and c.fb_id3 = f3.fb_id and c.fb_id4 = f4.fb_id and c.fb_id5 = f5.fb_id;
